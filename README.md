## Installation

``` bash
npm i @verts/axe
```

## Usage

``` vue
<template>
  <div id="app">
    <VueImageUploader
      xx=xx
    ></VueImageUploader>
  </div>
</template>

<script>
  import {VueImageUploader} from 'axe'
  export default {
    components: {
      VueImageUploader
    }
  }
</script>
```
